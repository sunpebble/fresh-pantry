# FreshPantry 四语言支持（zh-Hans / en / ja / fr）Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** iOS app（含 Widget、ShareExtension）、通知、报错、以及 364 道内置食谱全部支持简体中文、英语、日语、法语，跟随系统/每 app 语言设置。

**Architecture:** UI 与错误文案走语义化 key + String Catalog（`.xcstrings`，源语言 zh-Hans）；API 错误加稳定 `code` 字段由客户端映射译文；食谱由 recipe-pipeline 新增 AI 翻译阶段产出"按 id 的覆盖层"JSON 打包进 app，`RecipeLocalizer` 在加载后套用到任何来源（bundle/缓存/远程）的数据上。

**Tech Stack:** SwiftUI + String Catalog（xcstrings）、XcodeGen、Cloudflare Workers（TS + vitest）、recipe-pipeline（TS + valibot + vitest、Cloudflare AI chat-completions）。

**Spec:** `docs/superpowers/specs/2026-07-05-i18n-design.md`

## Global Constraints

- 语言集合固定为 `zh-Hans`（源）+ `en` + `ja` + `fr`；译文由实施者（AI）直接产出，无人工审校环节。
- 语义化 key 命名：`功能.组件.用途` 小驼峰（如 `tab.home`、`settings.backup.exportFailed`、`error.ai.quotaExhausted`）。带插值的 key 末尾跟占位符（`String.LocalizationValue` 插值自动生成，如 `"error.ai.authFailed \(status)"` → catalog key `error.ai.authFailed %lld`）。
- 每个 xcstrings 条目必须四语言齐全（`scripts/check_i18n.py` 强制）。en/fr 的数量文案用 plural variations；zh/ja 单一形式。
- 合法的中文字面量（解析映射表、`懒饭`/`下厨房` scheme 名、数据格式常量等**非 UI 展示**字符串）在行尾加 `// i18n:ignore` 标记豁免检查。
- iOS 工程是 XcodeGen 生成的：新增/移动文件后必须 `cd apps/ios && xcodegen generate`。
- Swift 6 strict concurrency（`SWIFT_STRICT_CONCURRENCY: complete`）、iOS 26 deployment target。
- 序列化用的 `en_US_POSIX` formatter（`JSONDate.swift`、`MealPlanEntry.swift`、`BackupView.swift:342` 的文件名日期）是数据格式，**不改**。
- 提交信息沿用仓库惯例：conventional commits + 中文主题（如 `feat(ios): ...`）。
- iOS 构建/测试命令（在 `apps/ios/` 下）：
  - 构建：`xcodebuild build -project FreshPantry.xcodeproj -scheme FreshPantry -sdk iphonesimulator -destination 'generic/platform=iOS Simulator' CODE_SIGNING_ALLOWED=NO`
  - 单测：`xcodebuild test -project FreshPantry.xcodeproj -scheme FreshPantry -destination 'platform=iOS Simulator,name=iPhone 17 Pro' -only-testing:FreshPantryTests`
- 现有测试全部保持通过：`npm run check`（仓库根，跑 api + recipe-pipeline 的 vitest）。

**Interfaces:**

- [ ] **Step 1: 写脚本**

```python
#!/usr/bin/env python3
"""i18n 闸门：
1) iOS 源码的字符串字面量里不允许出现中日文字符（注释除外；行尾 `// i18n:ignore` 豁免）；
2) 所有 .xcstrings 的每个 key 必须四语言（zh-Hans/en/ja/fr）齐全。
用法: check_i18n.py [swift源码路径...]   无参数 = 全量检查(源码+xcstrings)。
"""
import json
import re
import sys
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
IOS = ROOT / "apps" / "ios"
DEFAULT_DIRS = [IOS / "FreshPantry", IOS / "FreshPantryWidgets", IOS / "ShareExtension"]
LANGS = {"zh-Hans", "en", "ja", "fr"}
CJK = re.compile(r"[぀-ヿ一-鿿]")
STRING_LIT = re.compile(r'"(?:[^"\\\n]|\\.)*"')


def strip_comments(code: str) -> str:
    # 块注释换成等量换行，保住行号
    code = re.sub(r"/\*.*?\*/", lambda m: "\n" * m.group().count("\n"), code, flags=re.S)
    return re.sub(r"//[^\n]*", "", code)


def swift_offenders(dirs: list[Path]) -> list[str]:
    out = []
    for base in dirs:
        for f in sorted(base.rglob("*.swift")):
            if "build" in f.parts:
                continue
            raw_lines = f.read_text().splitlines()
            for i, line in enumerate(strip_comments(f.read_text()).splitlines(), 1):
                if i <= len(raw_lines) and "i18n:ignore" in raw_lines[i - 1]:
                    continue
                for m in STRING_LIT.finditer(line):
                    if CJK.search(m.group()):
                        out.append(f"{f.relative_to(ROOT)}:{i}: {m.group()[:80]}")
    return out


def xcstrings_offenders() -> list[str]:
    out = []
    for f in sorted(IOS.rglob("*.xcstrings")):
        if "build" in f.parts:
            continue
        catalog = json.loads(f.read_text())
        for key, entry in catalog.get("strings", {}).items():
            missing = LANGS - set(entry.get("localizations", {}))
            if missing:
                out.append(f"{f.relative_to(ROOT)}: '{key}' 缺 {sorted(missing)}")
    return out


def main() -> int:
    args = [Path(a).resolve() for a in sys.argv[1:]]
    offenders = swift_offenders(args or DEFAULT_DIRS)
    if not args:
        offenders += xcstrings_offenders()
    for line in offenders:
        print(line)
    print(f"\n{'FAIL' if offenders else 'OK'}: {len(offenders)} 处未本地化/缺翻译")
    return 1 if offenders else 0


if __name__ == "__main__":
    sys.exit(main())
```

```typescript
import { createHash } from 'node:crypto';
import * as v from 'valibot';
import { toJsonSchema } from '@valibot/to-json-schema';
import type { CleanRecipe } from '../clean/schema';
import { extractJson } from '../clean/cloudflare-enricher';
import { mapWithConcurrency } from '../util/pool';
import { CATEGORY_I18N } from './category-map';
import { RecipeOverlaySchema, type Lang, type RecipeOverlay, type TranslationCache } from './schema';

export const OVERLAY_JSON_SCHEMA = toJsonSchema(RecipeOverlaySchema, { errorMode: 'ignore' });

const LANG_NAMES: Record<Lang, string> = { en: 'English', ja: 'Japanese (日本語)', fr: 'French (français)' };

/** 只对可译字段取 hash:结构字段变化(图片/时长)不触发重译。 */
export function translatableHash(r: CleanRecipe): string {
  const subset = {
    name: r.name, description: r.description, category: r.category,
    steps: r.steps, tags: r.tags,
    ingredients: r.ingredients.map(({ name, unit, note }) => ({ name, unit, note })),
  };
  return createHash('sha256').update(JSON.stringify(subset)).digest('hex');
}

export function buildTranslatePrompt(r: CleanRecipe, lang: Lang): string {
  return [
    `Translate this Chinese home-cooking recipe into ${LANG_NAMES[lang]}.`,
    'Rules:',
    '- Keep the tone plain and friendly, like a well-kept kitchen note.',
    '- steps must have EXACTLY the same number of items as the source, same order.',
    '- ingredients must have EXACTLY the same number of items as the source, same order; translate name/unit/note only.',
    '- Use natural culinary units for the target language (e.g. 克→g, 只→whole/匹/entier as appropriate).',
    '- Do not add, drop, or merge content. Numbers and quantities stay as-is.',
    '- Return ONLY the JSON object.',
    '',
    JSON.stringify({
      name: r.name, description: r.description, category: r.category,
      steps: r.steps, tags: r.tags,
      ingredients: r.ingredients.map(({ name, unit, note }) => ({ name, unit, note })),
    }),
  ].join('\n');
}

export interface TranslateDeps {
  chat: (messages: { role: string; content: string }[], schemaName: string, jsonSchema: unknown) => Promise<string>;
  cache: TranslationCache;
  concurrency?: number;
  log?: (msg: string) => void;
}

export async function translateCorpus(
  recipes: CleanRecipe[], lang: Lang, deps: TranslateDeps,
): Promise<{ overlays: Record<string, RecipeOverlay>; cache: TranslationCache; failures: { id: string; error: string }[] }> {
  const log = deps.log ?? (() => {});
  const overlays: Record<string, RecipeOverlay> = {};
  const nextCache: TranslationCache = {};
  const failures: { id: string; error: string }[] = [];

  await mapWithConcurrency(recipes, deps.concurrency ?? 6, async (r) => {
    const hash = translatableHash(r);
    const hit = deps.cache[r.id];
    if (hit && hit.hash === hash) {
      overlays[r.id] = hit.overlay;
      nextCache[r.id] = hit;
      return;
    }
    try {
      const raw = await deps.chat(
        [{ role: 'user', content: buildTranslatePrompt(r, lang) }],
        'recipe_overlay', OVERLAY_JSON_SCHEMA,
      );
      const parsed = v.parse(RecipeOverlaySchema, JSON.parse(extractJson(raw)));
      if (parsed.steps.length !== r.steps.length) throw new Error(`steps 数量不齐: ${parsed.steps.length}≠${r.steps.length}`);
      if (parsed.ingredients.length !== r.ingredients.length) throw new Error(`ingredients 数量不齐`);
      const overlay: RecipeOverlay = { ...parsed, category: CATEGORY_I18N[r.category][lang] };
      overlays[r.id] = overlay;
      nextCache[r.id] = { hash, overlay };
      log(`translated ${r.id} → ${lang}`);
    } catch (err) {
      failures.push({ id: r.id, error: err instanceof Error ? err.message : String(err) });
    }
  });

  return { overlays, cache: nextCache, failures };
}
```

```swift
import Foundation

/// 打包进 app 的食谱译文覆盖层(howtocook.i18n.<lang>.json,id → 译文)。
/// 加载后套在【任何来源】(bundle/磁盘缓存/远程)的食谱上;中文界面不加载。
/// overlay 缺某 id(远程新增、翻译失败)→ 该条保持中文原文——离线优先的静默回退。
struct RecipeOverlayEntry: Decodable, Sendable {
    struct IngredientOverlay: Decodable, Sendable {
        let name: String
        let unit: String?
        let note: String?
    }
    let name: String
    let description: String
    let category: String
    let steps: [String]
    let tags: [String]
    let ingredients: [IngredientOverlay]
}

enum RecipeLocalizer {
    static let supported: Set<String> = ["en", "ja", "fr"]

    /// app 实际界面语言对应的 overlay 语言;中文或不支持的语言 → nil(用原文)。
    /// 用 Bundle.preferredLocalizations:跟随"设置→App→语言"的每 app 覆盖,与 UI 文案一致。
    static func overlayLanguage(
        preferred: [String] = Bundle.main.preferredLocalizations
    ) -> String? {
        for id in preferred {
            guard let code = Locale(identifier: id).language.languageCode?.identifier else { continue }
            if code == "zh" { return nil }
            if supported.contains(code) { return code }
        }
        return nil
    }

    /// 解码 bundle 里的 overlay;界面语言是中文或文件缺失/损坏 → nil(调用方直接用原文)。
    static func load(bundle: Bundle = .main) -> [String: RecipeOverlayEntry]? {
        guard let lang = overlayLanguage(),
              let url = bundle.url(forResource: "howtocook.i18n.\(lang)", withExtension: "json"),
              let data = try? Data(contentsOf: url),
              let overlay = try? JSONDecoder().decode([String: RecipeOverlayEntry].self, from: data)
        else { return nil }
        return overlay
    }

    /// 按 id 套用译文;食材按下标对齐,数量不齐则该条整体保持原文(防错位)。
    static func apply(_ overlay: [String: RecipeOverlayEntry], to recipes: [Recipe]) -> [Recipe] {
        recipes.map { recipe in
            guard let entry = overlay[recipe.id],
                  entry.ingredients.count == recipe.ingredients.count,
                  entry.steps.count == recipe.steps.count else { return recipe }
            var out = recipe
            out.name = entry.name
            out.description = entry.description
            out.category = entry.category
            out.steps = entry.steps
            out.tags = entry.tags
            out.ingredients = zip(recipe.ingredients, entry.ingredients).map { orig, tr in
                var ing = orig
                ing.name = tr.name
                if let unit = tr.unit { ing.unit = unit }
                if let note = tr.note { ing.note = note }
                return ing
            }
            return out
        }
    }
}
```

注意：`Recipe`/`RecipeIngredient` 若字段是 `let`，本任务把上述可译字段改为 `var`（域模型是 struct 值类型，改动安全）；若已有 lenient `Codable` 自定义实现，确认 `var` 不影响解码。
